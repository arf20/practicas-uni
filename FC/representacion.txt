float (simple precision IEEE 754)
	seeeeeeeemmmmmmmmmmmmmmmmmmmmmmm

	exponent 8 bit unsigned normal
	mantissa 23 bit que vale cada bit 1/2^i desde la izquierda

	s = (-1)^s
	E = e - S (127)
	M = 1.m
	N = (-1)^s * 2^(e - S) * 1.m
